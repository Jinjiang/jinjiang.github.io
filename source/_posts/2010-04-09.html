var list = document.head.getElementsByTagName('link');
for (var i = 0; i < list.length; i++) {
	if (list[i].rel == 'stylesheet') {
		document.head.removeChild(list[i]);
	}
}
var tempList = document.head.getElementsByTagName('style');
